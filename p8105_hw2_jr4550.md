``` r
unemployment_df = 
  read_csv("data/unemployment.csv", na = c("NA", ".","")) %>% 
  janitor::clean_names() %>% 
   pivot_longer(
    cols = jan:dec,
    names_to = "month",
    values_to = "unemployment_pct"
  ) |>
  mutate(
    year = as.integer(year),
    month = factor(month,
                   levels = tolower(month.abb),
                   labels = month.name),
    month = as.character(month)
  )
```
